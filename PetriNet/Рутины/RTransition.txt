routine RTransition[real T](InOut Place_Tr;Output Tr_Place)
	initial
		integer count;
		count:=0;
		array[1] of boolean pol;
		integer i;
		for i:=0 to 0 do
			pol[i]:=false;
		endf;
		array[2] of integer msg;

		Print "Инициализация вершины";
	endi
		
	event;
		msg:=StrToIntArray(message);
		if msg[0]=1 then
			Print "Получен запрос из полюса "+ToStr(msg[1]);
			if (pol[msg[1]-1]=false) then
				pol[msg[1]-1]:=true;
				count:=count+1;
				schedule Verify in 0;
			endif;
		else
		if msg[0]=3 then
			Print "Фишка получена через полюс "+ToStr(msg[1]);
			schedule Send in T;
			count:=0;
			for i:=0 to 0 do
				pol[i]:=false;
			endf;
		endif;
 			
		endif;
			
		
	ende

	event Send;
		msg[0]:=4;
		for i:=0 to 0 do
			out IntArrayToStr(msg) through Tr_Place;
			Print "Передача фишки через полюс "+ToStr(msg[1]);
		endf;
	ende

	event Verify;
		if (count=1) then
			Print "Получен запрос от всех вершин";
			msg[0]:=2;
			for i:=0 to 0 do
				msg[1]:=i;
				out IntArrayToStr(msg) through Place_Tr;
				Print "Отправляем ответ через полюс "+ToStr(msg[1]);
			endf;
		endif;
			
	ende
endrout
