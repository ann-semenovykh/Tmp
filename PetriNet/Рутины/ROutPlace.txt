routine ROutPlace[integer InitialNumMarks;real T](InOut Place_Tr)
	initial
		
		integer curpol;
		curpol:=1;
		schedule TryQuery in 0;
		array[2] of integer msg;
		Print "Инициализация вершины";
	endi
	
	event TryQuery;
		if InitialNumMarks>0 then
			msg[0]:=1;
			msg[1]:=curpol;
			out IntArrayToStr(msg) through Place_Tr;
			Print "Отправка запроса через полюс Place_Tr";
			schedule Wait in T;
		endif;
	ende

	event Wait;
		
			schedule TryQuery in 0;
	ende
	
	event;
		msg:=StrToIntArray(message);
		if msg[0]=2 then
			Print "Ответ получен через полюс Place_Tr";
			InitialNumMarks:=InitialNumMarks-1;
			msg[0]:=3;
			out IntArrayToStr(msg) through Place_Tr;
			Print "Отправляем фишку через полюс Place_Tr";
		endif;
		
	ende
	
endrout
