routine <имя_вершины>[real T](InOut Place_Tr[<кол-во вх.дуг>],Output Tr_Place[<кол-во вых. дуг>])
	initial
		integer count;
		count:=0;
		integer inarc,outarc;
		inarc:=<кол-во вых.дуг>;
		outarc:=<кол-во вх.дуг>;
		array[inarc] of boolean pol;
		for i:=1 to inarc do
			curpol[i]:=false;
		endf;
		array[2] of integer msg;

		Print "Инициализация вершины";
	endi
		
	event;
		msg:=StrToIntArray(message);
		if msg[0]=1 then
			Print "Получен запрос из полюса "+ToStr(msg[1]);
			if (curpol[ToInt(ToInt(msg[1])]=false) then
				curpol[ToInt(msg[1])]:=true;
				count:=count++;
				schedule Verify in 0;
			endif;
		else
		if msg[0]=3 then
			Print "Фишка получена через полюс "+ToStr(msg[1]);
			schedule Send in T;
			count:=0;
			for i:=1 to inarc do
				curpol[i]:=false;
			endf;
		endif;
 			
		endif;
			
		
	ende;

	event Send;
		msg[0]:=4;
		for i:=1 to outarc
			out IntArrayToStr(msg) through Tr_Place[i];
			Print "Передача фишки через полюс "+ToStr(msg[1]);
		endf;
	ende;

	event Verify;
		if (count=inarc) then
			Print "Получен запрос от всех вершин";
			msg[0]:=2;
			for i:=1 to inarc
				msg[1]:=i;
				out IntArrayToStr(msg) through Place_Tr[i];
				Print "Отправляем ответ через полюс "+ToStr(msg[1]);
			endf;
		endif;
			
	ende;
endrout.
